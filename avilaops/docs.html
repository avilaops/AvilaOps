Pipeline de Processamento de Imagens

python 
from arxis import AIVision, DataGrid
import asyncio

async def process_images_batch(image_urls):
    vision = AIVision(api_key="YOUR_KEY")
    results = []
    
    # Processar imagens em paralelo
    for url in image_urls:
        result = await vision.detect_objects_async(url)
        results.append({
            'url': url,
            'objects': len(result.objects),
            'labels': [obj.label for obj in result.objects]
        })
    
    # Salvar em DataFrame
    df = DataGrid.from_dict(results)
    df.to_parquet('vision_results.parquet')
    
    return df

# Executar pipeline
urls = ["img1.jpg", "img2.jpg", "img3.jpg"]
df = asyncio.run(process_images_batch(urls))
print(f"Processadas {len(df)} imagens")
